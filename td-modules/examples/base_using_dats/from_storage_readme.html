def data_sim(): new_dict = {} # create random vals for entry for each in range(op('grid1').numPoints): new_dict[each] = { 'sy' : tdu.rand(absTime.frame + each) } return new_dict # run the sim, and assign it to the varible data data = data_sim() parent().store('data', data)
